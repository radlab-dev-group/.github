Wdrożenie llm-router zajmuje mniej niż minutę

# Docker – najszybszy start
docker run -p 5555:8080 quay.io/radlab/llm-router:rc1
# Lub lokalnie
python3 -m venv .venv
source .venv/bin/activate
pip install .[api]
LLM_ROUTER_MINIMUM=1 python3 -m llm_router_api.rest_api
